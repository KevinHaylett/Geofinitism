import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

```
```
# Set precision and get digits of pi
mp.dps = 1000 # Number of decimal places for pi
pi_digits = mp.pi # Get pi as a high-precision number
pi_str = str(pi_digits).replace('.', '') # Remove decimal point
pi_array = np.array([int(d) for d in pi_str]) # Convert to array of digits
```
```
# Parameters for Takens' method
tau = 1 # Time delay (can experiment with values like 1, 2, 3)
embedding_dim = 3 # For 3D phase-space
n_points = len(pi_array) - (embedding_dim - 1 ) * tau # Number of delay vectors
```
```
# Create delay vectors
delay_vectors = np.zeros((n_points, embedding_dim))
for i in range(n_points):
delay_vectors[i] = [pi_array[i + k * tau] for k in range(embedding_dim)]
```
```
# Plot the attractor in 3D
fig = plt.figure(figsize=( 10 , 8 ))
ax = fig.add_subplot( 111 , projection='3d')
ax.plot(delay_vectors[:, 0 ], delay_vectors[:, 1 ], delay_vectors[:, 2 ],
color='purple', alpha=0.6, linewidth=0.5)
ax.set_xlabel('x(t)')
ax.set_ylabel('x(t + τ)')
ax.set_zlabel('x(t + 2τ)')
ax.set_title("Phase-Space Attractor of π's Digits (τ = {})".format(tau))
plt.show()

# Function to create delay vectors
def create_delay_vectors(data, tau, embedding_dim):
n_points = len(data) - (embedding_dim - 1 ) * tau
delay_vectors = np.zeros((n_points, embedding_dim))
for i in range(n_points):
delay_vectors[i] = [data[i + k * tau] for k in range(embedding_dim)]
return delay_vectors
